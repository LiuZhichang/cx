600 std::string target_log = CheckLogDir();

612 static std::string CheckLogDir() {
613 std::string cur_path = std::filesystem::current_path();
614 std::filesystem::path target_log(cur_path + "/" + m_log_dir);
615 if (!std::filesystem::exists(target_log)) {
616 std::filesystem::create_directory(target_log);
617 }
618
619 std::chrono::time_point now = std::chrono::system_clock::now();
620 time_t _now = std::chrono::system_clock::to_time_t(now);
621 std::stringstream ss;
622 ss << std::put_time(std::localtime(&_now), "%Y_%m_%d");
623
624 std::string cur_day_log = target_log.string() + "/" + ss.str() + ".log";
625 std::filesystem::path target_day_log(cur_day_log);
626 if (!std::filesystem::exists(target_day_log)) {
627 std::fstream fs;
628 fs.open(cur_day_log, std::ios_base::out);
629 fs.close();
630 }
631
632 return cur_day_log;
633 }
